% Full-width image block inside chapter files that are wrapped by multicols in main.tex.
% Usage: \FullWidthFigure[<width>]{<path>}{<caption>}
\newcommand{\FullWidthFigure}[3][\textwidth]{%
  \end{multicols}
  \newpage
  \begin{center}
    \begin{minipage}{\textwidth}
      \centering
      \IfFileExists{#2.tex}{%
        \input{#2}%
      }{%
        \includegraphics[width=#1]{#2}%
      }
      \captionsetup{type=figure,justification=centering}
      \captionof{figure}{#3}
    \end{minipage}
  \end{center}
  \newpage
  \begin{multicols}{2}%
}

% Column-width (half-page) image/figure INSIDE the multicols body.
% Flows in the current column (left or right, wherever the text currently is);
% non-floating, so it stays exactly where it is placed.
% Height is capped so image + caption never overflow into the footer.
% Usage: \ColumnFigure[<width>]{<path>}{<caption>}   (default width = column width)
\newcommand{\ColumnFigure}[3][\linewidth]{%
  \par\medskip
  \begin{center}
    \IfFileExists{#2.tex}{%
      \adjustbox{max width=#1,max totalheight=0.85\textheight}{\input{#2}}%
    }{%
      \adjustbox{max width=#1,max totalheight=0.85\textheight}{\includegraphics{#2}}%
    }
    \captionsetup{type=figure,justification=centering}
    \captionof{figure}{#3}
  \end{center}
  \par\medskip
}

% Feedback Capture Grid als 2x2-Raster ueber volle Breite.
% Argumente: #1 Positiv, #2 Optimieren, #3 Fragen, #4 Ideen
\newcommand{\FeedbackGrid}[4]{%
  {\renewcommand{\arraystretch}{1.35}%
  \setlength{\tabcolsep}{6pt}%
  \noindent\begin{tabularx}{\textwidth}{|>{\raggedright\arraybackslash}X|>{\raggedright\arraybackslash}X|}
    \hline
    \cellcolor{fbPositiv}\textbf{Positiv -- Was funktioniert gut?} &
    \cellcolor{fbOptimieren}\textbf{Optimieren -- Was kann verbessert werden?} \\
    \hline
    #1 & #2 \\
    \hline
    \cellcolor{fbFragen}\textbf{Fragen -- Welche Fragen kommen auf?} &
    \cellcolor{fbIdeen}\textbf{Ideen -- Wie koennte es aussehen?} \\
    \hline
    #3 & #4 \\
    \hline
  \end{tabularx}}%
}
