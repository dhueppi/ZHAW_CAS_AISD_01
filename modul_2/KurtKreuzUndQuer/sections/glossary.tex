% Glossarbegriffe
\newglossaryentry{doublediam}{
  name={Double Diamond},
  description={Vom British Design Council (2005) entwickeltes Modell des Design-Thinking-Prozesses, das vier Phasen (Discover, Define, Develop, Deliver) in zwei aufeinanderfolgenden Rauten visualisiert}
}

\newglossaryentry{designthinking}{
  name={Design Thinking},
  description={Nutzerzentrierter Innovations- und Probleml\"osungsansatz, der iteratives Vorgehen, Empathie f\"ur die Zielgruppe und experimentelles Prototyping verbindet}
}

\newglossaryentry{pov}{
  name={Point of View},
  description={Pr\"azise Problemformulierung im Design Thinking, die das noch unerf\"ullte Bed\"urfnis einer bestimmten Nutzergruppe beschreibt}
}

\newglossaryentry{challengedec}{
  name={Challenge Deconstruction},
  description={Methode zur Zerlegung einer Design Challenge in Kernbegriffe, deren Assoziationen, implizite Annahmen und Spannungsfelder, um ein gemeinsames Verst\"andnis im Team zu schaffen}
}

\newglossaryentry{stakeholdermap}{
  name={Stakeholder Map},
  description={Visualisierung aller relevanten Akteur:innen in konzentrischen Kreisen zur Identifikation von Einflussfaktoren, Partnerm\"oglichkeiten und Ber\"ucksichtigung von Nutzer:innen-\"okosystemen}
}

\newglossaryentry{deskresearch}{
  name={Desk Research},
  description={Systematische Analyse und Auswertung von bereits bestehenden Quellen, Studien und Daten zur evidenzgestützten Problemraumerschließung}
}

\newglossaryentry{designspace}{
  name={Design Space Map},
  description={Synthese des Wissensstands in drei Kategorien (Wir wissen, Wir nehmen an, Wir wissen nicht) zur Schärfung des Forschungsauftrags und Vorbereitung der nächsten Projektphase}
}

\newglossaryentry{modalwahl}{
  name={Modalwahl (Modalsplit)},
  description={Verteilung der zur\"uckgelegten Wege oder Kilometer auf verschiedene Verkehrsmittel (motorisierter Individualverkehr, öffentlicher Verkehr, Fußverkehr, Radverkehr)}
}

\newglossaryentry{leitfadeninterview}{
  name={Leitfadeninterview},
  description={Halbstandardisierte Interviewform, die einem flexibel anpassbaren Fragenleitfaden folgt und so vergleichbare Innensichten der Befragten erm\"oglicht, ohne den Gespr\"achsverlauf starr vorzugeben}
}

\newglossaryentry{empathymap}{
  name={Empathy Map},
  description={Synthesewerkzeug, das Aussagen und Beobachtungen einer Zielgruppe in die Quadranten Hear, See, Think \& Feel sowie Say \& Do ordnet, um Bed\"urfnisse und Spannungsfelder sichtbar zu machen}
}

\newglossaryentry{paingain}{
  name={Pain- und Gain-Map},
  description={Gegen\"uberstellung von Schmerzpunkten (Pains) und Gewinnen bzw. Motivatoren (Gains) der Zielgruppe zur Priorisierung von Hebeln f\"ur die sp\"atere L\"osungsentwicklung}
}
