% Glossarbegriffe
\newglossaryentry{doublediam}{
  name={Double Diamond},
  description={Vom British Design Council (2005) entwickeltes Modell des Design-Thinking-Prozesses, das vier Phasen (Discover, Define, Develop, Deliver) in zwei aufeinanderfolgenden Rauten visualisiert}
}

\newglossaryentry{designthinking}{
  name={Design Thinking},
  description={Nutzerzentrierter Innovations- und Probleml\"osungsansatz, der iteratives Vorgehen, Empathie f\"ur die Zielgruppe und experimentelles Prototyping verbindet}
}

\newglossaryentry{pov}{
  name={Point of View},
  description={Pr\"azise Problemformulierung im Design Thinking, die das noch unerf\"ullte Bed\"urfnis einer bestimmten Nutzergruppe beschreibt}
}

\newglossaryentry{challengedec}{
  name={Challenge Deconstruction},
  description={Methode zur Zerlegung einer Design Challenge in Kernbegriffe, deren Assoziationen, implizite Annahmen und Spannungsfelder, um ein gemeinsames Verst\"andnis im Team zu schaffen}
}

\newglossaryentry{stakeholdermap}{
  name={Stakeholder Map},
  description={Visualisierung aller relevanten Akteure in konzentrischen Kreisen zur Identifikation von Einflussfaktoren, Partnerm\"oglichkeiten und Ber\"ucksichtigung von Nutzer-\"okosystemen}
}

\newglossaryentry{deskresearch}{
  name={Desk Research},
  description={Systematische Analyse und Auswertung von bereits bestehenden Quellen, Studien und Daten zur evidenzgestützten Problemraumerschließung}
}

\newglossaryentry{designspace}{
  name={Design Space Map},
  description={Synthese des Wissensstands in drei Kategorien (Wir wissen, Wir nehmen an, Wir wissen nicht) zur Schärfung des Forschungsauftrags und Vorbereitung der nächsten Projektphase}
}

\newglossaryentry{modalwahl}{
  name={Modalwahl (Modalsplit)},
  description={Verteilung der zur\"uckgelegten Wege oder Kilometer auf verschiedene Verkehrsmittel (motorisierter Individualverkehr, öffentlicher Verkehr, Fußverkehr, Radverkehr)}
}

\newglossaryentry{leitfadeninterview}{
  name={Leitfadeninterview},
  description={Halbstandardisierte Interviewform, die einem flexibel anpassbaren Fragenleitfaden folgt und so vergleichbare Innensichten der Befragten erm\"oglicht, ohne den Gespr\"achsverlauf starr vorzugeben}
}

\newglossaryentry{empathymap}{
  name={Empathy Map},
  description={Synthesewerkzeug, das Aussagen und Beobachtungen einer Zielgruppe in die Quadranten Hear, See, Think \& Feel sowie Say \& Do ordnet, um Bed\"urfnisse und Spannungsfelder sichtbar zu machen}
}

\newglossaryentry{paingain}{
  name={Pain- und Gain-Map},
  description={Gegen\"uberstellung von Schmerzpunkten (Pains) und Gewinnen bzw. Motivatoren (Gains) der Zielgruppe zur Priorisierung von Hebeln f\"ur die sp\"atere L\"osungsentwicklung}
}

\newglossaryentry{persona}{
  name={Persona},
  description={Forschungsbasierter, fiktiver Archetyp eines Zielnutzers, der Bed\"urfnisse, Ziele, Einstellungen und Verhalten einer Nutzergruppe verdichtet und dem Team eine konsistente nutzerzentrierte Perspektive gibt}
}

\newglossaryentry{customerjourney}{
  name={Customer Journey},
  description={Visualisierung des gesamten Weges, den eine Persona bei der Nutzung eines Angebots durchl\"auft, von der ersten Idee bis zur Nachbetrachtung, inklusive Touchpoints, Pleasure- und Pain-Points sowie Emotionen je Phase}
}

\newglossaryentry{hmw}{
  name={How-might-we-Frage},
  description={Aus dem Point of View abgeleitete, l\"osungsoffene Impulsfrage (\enquote{Wie k\"onnten wir \dots}), die ein Bed\"urfnis in einen bearbeitbaren Gestaltungsauftrag \"ubersetzt und den Ideenraum der Develop-Phase aufspannt}
}

\newglossaryentry{kundentypologie}{
  name={Stimmt Kundentypologie},
  description={Von der Agentur Stimmt entwickeltes Modell, das Kunden anhand wiederkehrender Bed\"urfnis- und Verhaltensmuster in Grundtypen einteilt und so als strukturierte Grundlage f\"ur die Persona-Bildung dient}
}

\newglossaryentry{kopfstand}{
  name={Kopfstandmethode},
  description={Kreativtechnik des Reverse Brainstorming, die die Ausgangsfrage bewusst ins Gegenteil verkehrt (\enquote{Wie verhindern wir \dots?}), um blockierende Faktoren sichtbar zu machen und daraus L\"osungsans\"atze abzuleiten}
}

\newglossaryentry{prototyping}{
  name={Prototyping},
  description={Iteratives Erstellen einfacher, testbarer Vorabversionen einer L\"osung (Skizzen, Mockups, Kampagnenmaterial, Klick-Prototypen), um Ideen fr\"uh erfahrbar und mit Nutzenden \"uberpr\"ufbar zu machen}
}

\newglossaryentry{feedbackgrid}{
  name={Feedback Capture Grid},
  description={Strukturiertes Auswertungsraster f\"ur Nutzertests, das R\"uckmeldungen in vier Quadranten ordnet, Positives, Optimierungspunkte, offene Fragen und neue Ideen, um Beobachtungen je Prototyp vergleichbar zu verdichten}
}

\newglossaryentry{confirmationbias}{
  name={Confirmation Bias},
  description={Kognitive Verzerrung (Best\"atigungsfehler), bei der Informationen bevorzugt so gesucht, interpretiert und erinnert werden, dass sie die eigenen Erwartungen best\"atigen, w\"ahrend widersprechende Hinweise ausgeblendet oder abgewertet werden}
}

\newglossaryentry{haloeffekt}{
  name={Halo-Effekt},
  description={Kognitive Verzerrung (\"Uberstrahlungseffekt), bei der ein einzelnes hervorstechendes (meist positives) Merkmal einer Person oder Sache die Wahrnehmung und Bewertung ihrer \"ubrigen Eigenschaften \"uberstrahlt}
}

\newglossaryentry{ingroupbias}{
  name={Ingroup-Bias},
  description={Tendenz, Mitglieder der eigenen Gruppe (Ingroup) wohlwollender und positiver zu bewerten als Aussenstehende, auch ohne sachlichen Grund}
}
